    function getActionItems(member: any) {
      const items = []

      // Non-destructive actions group
      const nonDestructive = [
        {
          label: 'Reset Password',
          icon: 'i-lucide-key',
          onSelect: () => { selectedStaff.value = member; newPassword.value = ''; showPasswordReset.value = true },
        },
        {
          label: 'Pindah Cabang',
          icon: 'i-lucide-git-branch',
          onSelect: () => { selectedStaff.value = member; newBranchId.value = ''; showBranchReassign.value = true },
        },
      ]
      items.push(nonDestructive)

      // Status toggle (contextual)
      const destructive = []
      if (member.is_active) {
        destructive.push({
          label: 'Nonaktifkan',
          icon: 'i-lucide-user-x',
          color: 'error' as const,
          onSelect: () => { selectedStaff.value = member; showDeactivateConfirm.value = true },
        })
      } else {
        // Reactivate is NOT destructive — no confirmation needed
        items.push([{
          label: 'Aktifkan',
          icon: 'i-lucide-user-check',
          onSelect: () => handleReactivate(member),
        }])
      }

      destructive.push({
        label: 'Hapus',
        icon: 'i-lucide-trash-2',
        color: 'error' as const,
        onSelect: () => { selectedStaff.value = member; showDeleteConfirm.value = true },
      })
      items.push(destructive)

      return items
    }

    **Handler functions:**

    `handleCreate()`:
    - Set `creating.value = true`
    - Try: `await createStaff({ email: createForm.email, password: createForm.password, display_name: createForm.display_name || undefined, role: createForm.role, scope_type: createForm.scope_type as 'business' | 'branch', scope_id: createForm.scope_id })`
    - On success: `toast.add({ title: 'Staff berhasil dibuat', color: 'success', icon: 'i-lucide-check' })`, close slideover, reset form
    - On error: `toast.add({ title: e.data?.message || 'Gagal membuat staff', color: 'error', icon: 'i-lucide-alert-circle' })`
    - Finally: `creating.value = false`
    - Reset form: set all fields back to defaults (email='', password='', display_name='', role='cashier', scope_type='', scope_id='')

    `handleDelete()`:
    - Set `actionLoading.value = true`
    - Try: `await deleteStaff(selectedStaff.value.id)`
    - Toast: `'Staff berhasil dihapus'` success
    - Close modal: `showDeleteConfirm.value = false`
    - Catch: toast error
    - Finally: `actionLoading.value = false`

    `handleDeactivate()`:
    - Set `actionLoading.value = true`
    - Try: `await toggleStatus(selectedStaff.value.id, 'deactivate')`
    - Toast: `'Staff berhasil dinonaktifkan'` success
    - Close modal: `showDeactivateConfirm.value = false`
    - Catch: toast error
    - Finally: `actionLoading.value = false`

    `handleReactivate(member)`:
    - Try: `await toggleStatus(member.id, 'reactivate')`
    - Toast: `'Staff berhasil diaktifkan'` success
    - Catch: toast error

    `handlePasswordReset()`:
    - Set `actionLoading.value = true`
    - Try: `await resetPassword(selectedStaff.value.id, newPassword.value)`
    - Toast: `'Password berhasil diubah'` success
    - Close modal: `showPasswordReset.value = false`
    - Catch: toast error
    - Finally: `actionLoading.value = false`

    - Toast: `'Cabang berhasil diubah'` success
    - Close modal: `showBranchReassign.value = false`
    - Catch: toast error
    - Finally: `actionLoading.value = false`

    ```typescript
    function roleLabel(role: string) {
      const map: Record<string, string> = { owner: 'Pemilik', admin: 'Admin', cashier: 'Kasir' }
      return map[role] || role
    }

    function roleBadgeColor(role: string) {
      const map: Record<string, string> = { owner: 'primary', admin: 'warning', cashier: 'neutral' }
      return map[role] || 'neutral'
    }

    function scopeLabel(member: any) {
      if (member.scope_type === 'business') return 'Bisnis'
      const branch = branches.value.find((b: any) => b.id === member.scope_id)
      return branch?.name ?? 'Tanpa Cabang'
    }

    const branchItems = computed(() => branches.value.map((b: any) => ({ label: b.name, value: b.id })))
    const scopeTypeItems = [
      { label: 'Bisnis', value: 'business' },
      { label: 'Cabang', value: 'branch' },
    ]
    const roleItems = [
      { label: 'Admin', value: 'admin' },
      { label: 'Kasir', value: 'cashier' },
    ]

    function staffDisplayName(member: any) {
      return member.display_name || member.email || member.auth_user_id
    }
    ```
